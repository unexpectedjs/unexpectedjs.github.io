    Asserts that a promise is rejected with a specific reason (error):

    var promiseThatWillBeRejectedWithAReason = new Promise(function(
      resolve,
      reject
    ) {
      setTimeout(function() {
        reject(new Error('Oh dear'));
      }, 10);
    });
     
    return expect(
      promiseThatWillBeRejectedWithAReason,
      'to be rejected with error satisfying',
      new Error('Oh dear')
    );

    The expected reason will be matched against the rejection reason with to satisfy semantics, so you can pass any of the values supported by to satisfy:

    var promiseThatWillBeRejectedWithAReason = new Promise(function(
      resolve,
      reject
    ) {
      setTimeout(function() {
        reject(new Error('Oh dear'));
      }, 10);
    });
     
    return expect(
      promiseThatWillBeRejectedWithAReason,
      'to be rejected with error satisfying',
      /dear/
    );
